## Usage

```tsx
import { OtpStyledInput } from "@/components/extension/otp-input";
```

```tsx
<OtpStyledInput
  numInputs={5}
  inputType="number"
  value={value}
  onChange={onChange}
/>
```

```tsx showLineNumbers {1 , 3 , 16-21  }

        {...}
     </Form>
  )
}
```
