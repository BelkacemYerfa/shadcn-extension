## Usage

```tsx
import { SmartDateTimeInput } from "@/components/extension/smart-date-time-input";
```

```tsx
<SmartDateTimeInput
  name="datetime"
  value={Date.now()}
  onChange={(e) => setValue(e.target.value)}
  placeholder="e.g. tomorrow at 3pm"
/>
```
