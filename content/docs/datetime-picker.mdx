## Usage

```tsx
import { DatetimePicker } from "@/components/extension/datetime-picker";
```

```tsx
<DatetimePicker
  value={Date.now()}
  onChange={(date) => console.log(date)}
  config={
    (["months", "days", "years"], ["hours", "minutes", "seconds", "am/pm"])
  }
/>
```
